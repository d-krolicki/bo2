procedure algo(Shop, iterationStop, changesInPopulation, populationSize, penVal, CROSSOVER_PROBABILITY, MUTATION_PROBABILITY_SWAP_AMOUNT, MUTATION_PROBABILITY_VISIT_ORDER, MUTATION_PROBABILITY_ADD_OR_REMOVE_POINT, MUTATION_PROBABILITY_SUBTRACT_AMOUNT)
p = new Population(shop, populationSize)
initialize initial-population
calculate cost-function
sort p
for i in range(iterationStop) do
new_p ← empty list
for s in range(populationSize//2) do
crossP ← random(0,1)
if crossP < CROSSOVER_PROBABILITY do
par1 ← ROULETTE_SELECTION()
par2 ← ROULETTE_SELECTION()
{c1, c2} ← execute CROSSOVER(par1, par2)
insert c1 into new_p
insert c2 into new_p
end if
end for
p ← new_p
fill p
for t in range(populationSize) do
mutation1P ← random(0,1)
mutation2P ← random(0,1)
mutation3P ← random(0,1)
mutation4P ← random(0,1)
if mutation1P < MUTATION_PROBABILITY_SWAP_AMOUNT do
execute MUTATION_SWAP_AMOUNT
end if
if mutation2P < MUTATION_PROBABILITY_VISIT_ORDER do
execute MUTATION_VISIT_ORDER
end if
if mutation3P < MUTATION_PROBABILITY_ADD_OR_REMOVE_POINT do
execute MUTATION_ADD_OR_REMOVE_POINT
end if
if mutation4P < MUTATION_PROBABILITY_SUBTRACT_AMOUNT do
execute MUTATION_SUBTRACT_AMOUNT
end if
calculate cost_function
sort p
end for
end for
return p[0]
end procedure
