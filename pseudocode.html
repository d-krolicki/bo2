GLOBAL CROSSOVER_PROBABILITY
GLOBAL MUTATION_PROBABILITY_SWAP_AMOUNT
GLOBAL MUTATION_PROBABILITY_VISIT_ORDER
GLOBAL MUTATION_PROBABILITY_ADD_OR_REMOVE_POINT
GLOBAL MUTATION_PROBABILITY_SUBTRACT_AMOUNT

procedure algo(Shop, iterationStop, changesInPopulation, populationSize, penVal)
p = new Population(shop, populationSize)
initialize initial-population
calculate cost-function
while iterationStop > 0 do
c ← changesInPopulation
while iterationStop > 0 do
calculate cost-function
mutation1P ← random(0,1)
mutation2P ← random(0,1)
mutation3P ← random(0,1)
mutation4P ← random(0,1)
crossoverP ← random(0,1)
if mutation1P < MUTATION_PROBABILITY_SWAP_AMOUNT do
execute MUTATION_SWAP_AMOUNT
sort p
if mutation2P < MUTATION_PROBABILITY_VISIT_ORDER do
execute MUTATION_VISIT_ORDER
sort p
if mutation3P < MUTATION_PROBABILITY_ADD_OR_REMOVE_POINT do
execute MUTATION_ADD_OR_REMOVE_POINT
sort p
if mutation4P < MUTATION_PROBABILITY_SUBTRACT_AMOUNT do
execute MUTATION_SUBTRACT_AMOUNT
sort p
if crossoverP < CROSSOVER_PROBABILITY do
execute CROSSOVER
sort p
calculate cost-function
c ← c - 1
end while
iterationStop ← iterationStop - 1
end while
return p[0]
end procedure
